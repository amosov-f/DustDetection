    \section{Постановка задачи} 

        \begin{frame}{Постановка задачи}
            Построить карту градиентов покраснения звезд вдоль лучей зрения в окрестности Солнца \\
            
            Используемые данные:    \\
            звездные каталоги Hipparcos и Hipnewcat    \\
            ($98827$ звезд)
            \begin{itemize}
                \item положения (Hipnewcat)
                \item параллаксы (Hipnewcat)
                \item фотометрия (Hipparcos)
                \item спектральные классы и классы светимости (Hipparcos)
            \end{itemize}
        \end{frame}
        
        \begin{frame}{Межзвездное покраснение звезд}
            $$
                E_{B - V} = (B - V)_{obs} - (B - V)_{int}
            $$
            Источники:
            \begin{itemize}
                \item $(B - V)_{obs}$ --- каталог Hipparcos
                \item $(B - V)_{int}$ --- {\small J. Binney et al. <<Galactic Astronomy>>}
            \end{itemize}
            
            { \footnotesize
                Пример (звезда HIP 44800),
                \begin{itemize}
                    \item У нее в каталоге $(B - V)_{obs} = 0.535^m$
                    \item Класс F7V    
                        \begin{tabular} {| c | c | c | c |}
                            \hline
                               &    $\cdots$    &    V    &    $\cdots$    \\
                            \hline
                            $\cdots$    &    &    &    \\
                            \hline
                            F7    &    &    0.493    &        \\
                            \hline
                            $\cdots$     &     &     &      \\
                            \hline
                        \end{tabular}
                    поэтому $(B - V)_{int} = 0.493^m$
                    \item Покраснение $0.535^m - 0.493^m = 0.042^m$
                    \item Между нами и звездой пыли на $0.042^m$
                \end{itemize}
            }
            
        \end{frame}
