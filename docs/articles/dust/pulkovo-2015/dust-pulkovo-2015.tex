    \section{Постановка задачи} 

        \begin{frame}{Постановка задачи}
            Построить карту градиентов покраснения звезд вдоль лучей зрения в окрестности Солнца \\
            ~\\
            Используемые данные: звездные каталоги    \\
            Hipparcos (Perryman et al., 1997) и    \\
            Hipnewcat (van Leeuwen, 2007)    \\
            ($98827$ звезд)
            \begin{itemize}
                \item положения (Hipnewcat)
                \item параллаксы (Hipnewcat)
                \item фотометрия (Hipparcos)
                \item спектральные классы и классы светимости (Hipparcos)
            \end{itemize}
        \end{frame}
        
        \begin{frame}{Межзвездное покраснение звезд}
            $$
                E_{B - V} = (B - V)_{obs} - (B - V)_{int}
            $$
            Источники:
            \begin{itemize}
                \item $(B - V)_{obs}$ --- каталог Hipparcos
                \item $(B - V)_{int}$ --- {\small J. Binney et al. <<Galactic Astronomy>>}
            \end{itemize}
            
            { \footnotesize
                Пример (звезда HIP 44800),
                \begin{itemize}
                    \item У нее в каталоге $(B - V)_{obs} = 0.535^m$
                    \item Класс F7V    
                        \begin{tabular} {| c | c | c | c |}
                            \hline
                               &    $\cdots$    &    V    &    $\cdots$    \\
                            \hline
                            $\cdots$    &    &    &    \\
                            \hline
                            F7    &    &    0.493    &        \\
                            \hline
                            $\cdots$     &     &     &      \\
                            \hline
                        \end{tabular}
                    поэтому $(B - V)_{int} = 0.493^m$
                    \item Покраснение $0.535^m - 0.493^m = 0.042^m$
                    \item Между нами и звездой пыли на $0.042^m$
                \end{itemize}
            }
            
        \end{frame}

        \begin{frame}{Задача классификации звезд по классам светимости}
            \begin{itemize}
                \item Алгоритм классификации: метод опорных векторов (Support Vector Machines, SVM)
                \item ПО: WEKA, Java--библиотека для ML
                \item 
                    \begin{minipage}{0.6\textwidth}
        			        Факторы: показатель цвета, \\
        			        абсолютная звездная величина
                    \end{minipage}
        		        \hfill
        		        \begin{minipage}{0.3\textwidth}
        			        \includegraphics[width=3.5cm]{gr-example.eps}
        		        \end{minipage}
                
                \item Класс: класс светимости \\
                III или V --- бинарный классификатор
            \end{itemize}
        \end{frame}        
        
        \begin{frame}{Классификатор}
            \begin{center}
                \includegraphics[width=9cm]{classify.eps}    \\
                III, V класс --- 39807 звезд
            \end{center}       
            $$ -3.0752 \cdot (B - V) + 0.4485 \cdot M + 1.4793 = 0 $$      
        \end{frame}
        
        \begin{frame}{Классификатор (продолжение)}
            \begin{center}
                \includegraphics[width=9cm]{main-lumin.eps}    \\
                Обучающее множество на небесной сфере
            \end{center} 
            \begin{center}
                \begin{tabular} {| c | c | c |}
                    \hline
                    Класс    &    Точность   &    Полнота     \\
                    \hline
                    III      &    95\%   &   89\%     \\
                    \hline
                    V        &    91\%   &    96\%         \\
                    \hline
                \end{tabular}
            \end{center}
        \end{frame}       
        
        \begin{frame}{Другие классификаторы}
            Алгоритм классификации: bagging \\
            Факторы: $B-V, M, m$
            \begin{itemize}
                \item Отделили III класс от V в области \\ 
                $0.3 < B-V < 0.6, m < 6.5$    \\ 
                $p=63\%, r=44\%$
                \item Отделили II класс от III и V в области    \\
                $0.3 < B-V < 0.9, M < 1$    \\
                $p=52\%, r=36\%$
            \end{itemize}
        \end{frame}
